var mongoose = require('mongoose')
var Schema = mongoose.Schema

var rekordSchema = new Schema({
  title: String,
  slug: String,
  userId: String,
  content: Array,
  created: Date,
  updated: Date,
  dateFrom: Date,
  dateTo: Date,
  share: {
    googlep: Number,
    twitter: Number,
    facebook: Number
  },
  status: String,
  theme: {
    coverOpacity: Number,
    coverThumbSmall: String,
    coverThumb: String,
    coverImage: String
  },
  deleted: Boolean,
  active: Boolean,
  place: {
    name: String,
    lon: String,
    lat: String
  },
  matchId: String
})

mongoose.model('pages', rekordSchema)
