const restify = require('restify')

// Handles
const storyHndl = require('../handles/story.handles')
// Services
const log = require('../services/log').log
// Middleware
const responseMiddleware = require('../middleware/response')

exports.myStories = async (req, res, next) => {
  try {
    let user = req.effectiveUser
    let filters = req.params.filters
    let status = req.params.status
    let search = req.params.search

    let stories = await storyHndl.myStories(user.id, filters, status, search)

    responseMiddleware.paginateV3(req, res, stories)

  } catch (err) {
    log.error(err)
    return next(restify.InternalServerError(err))
  }
}
