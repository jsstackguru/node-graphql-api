// dependencies
const config = require('../../config')
var PostageApp = require('postageapp')
var postage = new PostageApp(config.postageapp.API_ID)
var uuid = require('node-uuid')

/**
 * Send email via Postageapp service
 *
 * @param {Object} data
 */
const send = function (data) {
  data.uid = uuid.v4()
  postage.sendMessage(data)
}

module.exports.send = send

/**
 * Author's activation email with activation link
 *
 * @param {Object} author Author's data (email, name, username)
 */
module.exports.activation = (author, token) => {
  send({
    recipients: author.email,
    template: 'activate_user',
    variables: {
      name: author.name,
      activation_url: config.web_url + '/activate-user/' + token
    }
  })
}

/**
 * Send email for reset password
 *
 * @param {object} user     user who need to receive an email
 * @param {string} code     code for validation
 */
module.exports.resetPassword = (user, code) => {
  send({
    recipients: user.email,
    template: 'reset_password',
    variables: {
      name: user.name,
      activation_url: config.web_url + '/resetpass/' + user.id + '/' + code
    }
  })
}
