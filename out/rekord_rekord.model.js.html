var mongoose = require('mongoose')
var Schema = mongoose.Schema

var rekordSchema = new Schema({
  title: String,
  slug: String,
  status: String,
  created: Date,
  views: Number,
  theme: Object,
  deleted: Boolean,
  active: Boolean,
  banInPublic: Boolean,
  collaborators: [{
    // author: { type: Schema.Types.ObjectId, ref: 'authors' },
    author: { type: String, ref: 'authors' },
    edit: Boolean
  }],
  updated: Date,
  spam: Number,
  featured: Boolean,
  sharedLink: Boolean,
  // author: { type: Schema.Types.ObjectId, ref: 'authors' },
  author: { type: String, ref: 'authors' },
  // pages: [{ type: Schema.Types.ObjectId, ref: 'pages' }],
  pages: [{ type: String, ref: 'pages' }]
})

/**
 * Get list of stories by query and page match
 * @param {Object} query Query for stories filter
 */
rekordSchema.statics.getList = (query, pageMatch) => {
  return new Promise((resolve, reject) => {
    // check does email is already taken
    let rekordModel = mongoose.model('stories')
      .find(query)
      .populate({ path: 'author', select: 'id name username avatar' })
      .populate({ path: 'pages', match: pageMatch })

    rekordModel.then((stories) => {
      resolve(stories)
    })
      .catch((err) => {
        console.log(err)
        reject(err)
      })
  })
}
